program:
    expression*

expression:
    | binding
    | function_def
    | if_expr
    | binary_expr

binding:
    IDENTIFIER ":=" expression

function_def:
    IDENTIFIER "|->" expression

if_expr:
    "if" expression "then" expression "else" expression

binary_expr:
    term (("+" | "-") term)*

function_call:
    IDENTIFIER "(" expression ")"

term:
    factor (("*" | "/" | ">") factor)*

factor:
    ("-" ?) unary_argument

unary_argument:
    | NUMBER
    | IDENTIFIER
    | function_call
    | "(" expression ")"

IDENTIFIER:
    [a-zA-Z_][a-zA-Z0-9_]*

NUMBER:
    ("-" ?) DIGITS "." (DIGITS ?)

DIGITS:
    [0-9]+
